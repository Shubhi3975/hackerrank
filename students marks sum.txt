

//Complete the following function.

int marks_summation(int* marks, int number_of_students, char gender) {
  //Write your code here.
  int sum =0,i,j;
  if(gender == 'b')
  {
      // for boys
      for(i =0;i<number_of_students;i=i+2)
      {
          sum = sum+marks[i];
      }
  }
  
   //for girls
 else if(gender == 'g')
{
    for(j=1;j<number_of_students;j=j+2)
    {
        sum=sum+marks[j];
    }
}   
 return sum;
    
}
